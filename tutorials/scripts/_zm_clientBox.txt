function registerBoxWeapon( name, dualWield )
{
	if( !isDefined( name ))
		return;

	isDualWield = false;

	if(isDefined( dualWield ))
		isDualWield = dualWield;

	weapon = GetWeapon( name );
	model = GetWeaponWorldModel( weapon );
	
	AddZombieBoxWeapon(weapon, model, isDualWield);
}
